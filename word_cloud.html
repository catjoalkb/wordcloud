import os

def import_file(file_path):
    lyrics = ''
    
    for f in os.listdir(file_path):
        abs_path = os.path.join(file_path, f)
        
        if os.path.isfile(abs_path):
            print('loading {}'.format(abs_path))
            
            lyrics += open(abs_path, 'r').read()
            
            lyrics += '\n'
    print('done loading')
    return lyrics
